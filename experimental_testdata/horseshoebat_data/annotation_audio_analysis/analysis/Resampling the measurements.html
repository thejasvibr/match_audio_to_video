This notebook will deal with the resampling and comparison of single and multi bat context audio segments.

Each video annotation was matched with a corresponding length of audio, and each annotation also had the number of bats seen in the video. In the previous notebook ('Split measuring all valid...'), each annotation audio was split into 50ms segments and various measurements were carried out on them (rms, -20dB frequency, dominant frequency).

In the current notebook I will now load the previous measurements of segments that had a non-negligible rms (+20dB above the level calculated for silent audio segments). These measurements contain measurement values for multiple segments from the same audio file. Comparing the measurements from all segments together may lead to spurious results because of the extent of pseudo-replication in the data. This is because, not only are the segments from the same audio file, but even many of the audio files are from the same hour.

Reducing the extent of pseudo-replication in the natural observations¶

Resampling to take out the measurements from one segment per audio file will help reduce the extent of pseudo-replication. After this, another exercise that could also be carried out is to perform some kind of N-fold cross-validation to understand if the obtained results are robust to the exact dataset used.

Characterising the bootstrapped data¶

After reducing the extent of pseudo-replication a little bit through the process of bootstrapping, the data is then summarised by different measures, for instance, the median, the first quartile(25%ile), median, and third quartile (75%ile). For each round of bootstrappping, one segment per audio file is chosen for all files, and the summary data is stored. After N such rounds, we will have the summary data distribution, which can then be compared between the single and multi bat groups.

def split_into_single_and_multi(df):
    '''
    '''
    single_bat = df[df['num_bats']==1].reset_index(drop=True)
    multi_bat  = df[df['num_bats']>1].reset_index(drop=True)
    return single_bat, multi_bat

def extract_one_measurement(df,measurement_name):
    '''
    '''
    one_measurement = df[df['measurement']==measurement_name].reset_index(drop=True)
    return one_measurement

def calc_delta_median(df, measurement_name, proc_fun=lambda X:X):
    '''Accepts a DataFrame with both single and multi bat segments in them 
    and then calculates the difference in the median between the single 
    and multi-bat measurements. 
    
    Specifically, it calculates the median(multi_bat)-median(single_bat)
    '''
    single, multi = extract_single_multi_by_measurement(df, measurement_name)
    multi_median = np.nanmedian(proc_fun(multi['value']))
    single_median = np.nanmedian(proc_fun(single['value']))
    delta_median = multi_median-single_median
    return delta_median

def calc_single_multi_median(df, measurement_name, proc_fun=lambda X:X):
    '''
    '''
    single, multi = extract_single_multi_by_measurement(df, measurement_name)
    multi_median = np.nanmedian(proc_fun(multi['value']))
    single_median = np.nanmedian(proc_fun(single['value']))
    return single_median, multi_median


def calc_single_multi_summary(df, measurement_name, summary_fn, proc_fun:lambda X:X):
    '''
    '''
    single, multi = extract_single_multi_by_measurement(df, measurement_name)
    multi_summary =  summary_fn(proc_fun(multi['value']))
    single_summary = summary_fn(proc_fun(single['value']))
    return single_summary, multi_summary

def extract_single_multi_by_measurement(df, measurement_name):
    '''
    '''
    by_measurements = df[df['measurement']==measurement_name].reset_index(drop=True)
    single_rms, multi_rms = by_measurements[by_measurements['num_bats']==1], by_measurements[by_measurements['num_bats']>1]
    return single_rms, multi_rms

def make_inspection_and_comparison_plot(df, measurement_name, process_fn=lambda X: X):
    '''
    '''
    single, multi = extract_single_multi_by_measurement(df, measurement_name)
    plt.figure(figsize=(8,4))
    plt.violinplot([process_fn(single['value'].to_numpy()), process_fn(multi['value'].to_numpy())],
                                                                   [0,1],
                                                          showmedians=True,
                                                            quantiles=[[.25,.75], [.25,.75]]);    
    plt.xticks([0,1],['single','multi'])
    plt.ylabel(measurement_name, fontsize=12)

def resample_one_segment_from_each_audio(df):
    '''
    Parameters
    ----------
    df : pd.DataFrame
        DataFrame with the following compulsory columns
        file_name, segment_number
    Returns 
    -------
    one_seg_per_file : pd.DataFrame
        A subset of all rows, but with only one segment per file chosen. 
    '''
    by_filename = df.groupby(['file_name'])
    resampled_data = list(map(choose_one_segment, by_filename))
    one_seg_per_file = pd.concat(resampled_data).reset_index(drop=True)
    return one_seg_per_file

def choose_one_segment(filename_and_subdf):
    filename, sub_df = filename_and_subdf
    # if there are >1 valid segments in the audio file, then select just one randomly
    segments = sub_df['segment_number'].tolist()
    if len(segments) == 1:
        return sub_df
    else:
        one_segment = int(np.random.choice(segments, 1))
        return sub_df[sub_df['segment_number']==one_segment]

def shuffle_values_between_single_and_multi(df, measurement_name):
    '''
    '''
    single, multi = extract_single_multi_by_measurement(df, measurement_name)
    single_rows, multi_rows = single.shape[0], multi.shape[0]
    # swap values between single and multi bat annotations
    all_values = np.concatenate([single['value'], multi['value']]).flatten()
    np.random.shuffle(all_values)
    single['value'] = all_values[:single_rows]
    multi['value'] = all_values[single_rows:]
    return single, multi

def make_shuffled_df(df, measurement_name):
    shuffled_single, shuffled_multi = shuffle_values_between_single_and_multi(df, measurement_name)
    joined = pd.concat([shuffled_single, shuffled_multi]).reset_index(drop=True)
    return joined

def calculate_observed_and_shuffled_overlap(df, measurement_name, summary_fn, proc_fn, num_shuffles=500):
    '''
    '''
    observed_shuffled_BC = Parallel(n_jobs=4)(delayed(get_observed_and_shuffled)(df, measurement_name, summary_fn, proc_fn) for i in tqdm.tqdm(range(num_shuffles)))
    observed = [each[0] for each in observed_shuffled_BC]
    shuffled =  [each[1] for each in observed_shuffled_BC]

    return observed, shuffled
        

def get_observed_and_shuffled(df, measurement_name, summary_fn, proc_fn):
    one_segment = resample_one_segment_from_each_audio(df)
    single_bat_measures, multi_bat_measures = calc_single_multi_summary(one_segment, 
                                                                      measurement_name, 
                                                                          summary_fn, proc_fn)
    this_run_BC, _ = shuffle_overlap.calculate_overlap(single_bat_measures, multi_bat_measures)

    # now do the data shuffling
    shuffled_segment = make_shuffled_df(one_segment, measurement_name)
    single_bat_shufmeasures, multi_bat_shufmeasures = calc_single_multi_summary(shuffled_segment, 
                                                                          measurement_name, 
                                                                          summary_fn, proc_fn)
    this_run_shufBC, _ = shuffle_overlap.calculate_overlap(single_bat_shufmeasures, multi_bat_shufmeasures)
    return this_run_BC, this_run_shufBC
